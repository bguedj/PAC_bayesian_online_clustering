\section{Notation}

\begin{frame}[fragile]{Notation}
$(x_{t})_{1:T}$ : online dataset, where $x_{t} \in \mathbb{R}^{d}$\\
$K_{t}$: nb of clusters\\
$\hat{c}_{t} = (\hat{c}_{t,1}, \hat{c}_{t,2},...\hat{c}_{t,K_{t}})$ : clusters location, depending on the past information $(x_{s})_{1:(t-1)}$ and $(\hat{c}_{s})_{1:(t-1)} $\\
\smallbreak
When $x_{t}$ is newly revealed, the instantaneous loss:\\
$\ell(\hat{c}_{t}, x_{t}) = \underset{1 \leq k \leq K_{t}}{min} |\hat{c}_{t,k} - x_{t}|^{2}_{2}$\\
\end{frame}

\begin{frame}[fragile]{Notation}
$\mathscr{C} = \bigcup^{p}_{k=1}$\\
q: discrete probability distribution on the set [1,p]:={1,..p}\\
for any k $\in$ [1,p], let $\pi_{k}$, the probability distribution on $\mathbb{R}^{dk}$\\
\smallbreak
For any c $\in \mathscr{C}$, we define $\pi(c)$, as\\
$\pi(c)= \sum_{k \in [1,p]} q(k) \mathbbm{1}_{\{ c\in\mathbb{R}^{dk}\}} \pi_{k}(c)$
\end{frame}

\begin{frame}[fragile]{Notation}
$c \in \mathscr{C}$ a partition of $\mathbb{R}^{d}, \pi \in \mathbb{P}(\mathscr{C})$ a quasi prior over this set\\
$\lambda >0$ : inverse temperature parameter\\
\smallbreak
At each time t, we observe $x_{t}$ and a random partition $\hat{c}_{t+1} \in \mathscr{C}$ is sampled from the quasi-posterior:\\
$d\hat{\rho}_{t+1} \propto \exp{\{ -\lambda S_{t}(c) \}}
d\pi(c)$\\
\smallbreak
Cumulative loss:\\
$S_{t}(c) = S_{t-1}(c) + \ell (c, x_{t}) + \frac{\lambda}{2}
\{ \ell(c, x_{t}) - \ell (\hat{c}_{t},  x_{t}) \}^{2}$\\
\end{frame}
